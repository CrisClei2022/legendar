    <script>
        let currentSegment = 0;
        let repeatCount = 1;
        let autoContinue = true;
        let segments = [];
        let videoPlayer = document.getElementById('video-player');

        document.getElementById('upload-form').addEventListener('submit', function(event) {
            event.preventDefault();
            const videoFile = document.getElementById('video-upload').files[0];
            const srtFile = document.getElementById('srt-upload').files[0];

            if (videoFile && srtFile) {
                const videoSource = document.getElementById('video-source');
                videoSource.src = URL.createObjectURL(videoFile);
                videoPlayer.load();

                const reader = new FileReader();
                reader.onload = function(e) {
                    const srtContent = e.target.result;
                    segments = generateLinks(srtContent);
                };
                reader.readAsText(srtFile);
            }
        });

        function generateLinks(srtContent) {
            const lines = srtContent.split('\n');
            let linksHTML = '';
            let currentTimestamp = 0;
            let segments = [];

            lines.forEach((line, index) => {
                if (line.trim().match(/^\d+$/)) {
                    return;
                }

                if (line.includes('-->')) {
                    currentTimestamp = parseTimestamp(line.split(' --> ')[0]);
                    segments.push(currentTimestamp);
                } else if (line.trim() !== '' && !line.includes('-->')) {
                    linksHTML += `<a class="sentence" href="#" data-time="${currentTimestamp}">${line.trim()}</a><br>`;
                }
            });

            document.getElementById('links-container').innerHTML = linksHTML;
            return segments;
        }

        function parseTimestamp(timestamp) {
            const parts = timestamp.split(':');
            const hours = parseInt(parts[0], 10);
            const minutes = parseInt(parts[1], 10);
            const seconds = parseFloat(parts[2].replace(',', '.'));
            return hours * 3600 + minutes * 60 + seconds;
        }

        document.getElementById('links-container').addEventListener('click', function(event) {
            if (event.target && event.target.className === 'sentence') {
                event.preventDefault();
                const time = parseFloat(event.target.getAttribute('data-time'));
                playSegment(time);
            }
        });

        function playSegment(time) {
            videoPlayer.pause();
            videoPlayer.currentTime = time;
            videoPlayer.play().catch(error => {
                console.error('Erro ao tentar reproduzir o vídeo:', error);
            });

            currentSegment = segments.indexOf(time);
        }

        document.getElementById('reproduzir-anterior').addEventListener('click', function() {
            if (currentSegment > 0) {
                currentSegment--;
                playSegment(segments[currentSegment]);
            }
        });

        document.getElementById('repetir-atual').addEventListener('click', function() {
            if (segments.length > 0) {
                playSegment(segments[currentSegment]);
            }
        });

        document.getElementById('reproduzir-proximo').addEventListener('click', function() {
            if (currentSegment < segments.length - 1) {
                currentSegment++;
                playSegment(segments[currentSegment]);
            }
        });

        document.getElementById('repeat-count').addEventListener('change', function() {
            repeatCount = parseInt(this.value, 10);
        });

        document.getElementById('auto-continue').addEventListener('change', function() {
            autoContinue = this.checked;
        });

        videoPlayer.addEventListener('ended', function() {
            if (repeatCount > 1) {
                repeatCount--;
                if (autoContinue && repeatCount > 0) {
                    this.currentTime = segments[currentSegment]; // Seek to the current segment
                    this.play();
                }
            }
        });
